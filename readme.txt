This is the common directory structure for a Flask application
your_flask_app/
|-- app/
|   |-- static/
|   |   |-- css/
|   |   |-- js/
|   |   |-- images/
|   |
|   |-- templates/
|   |   |-- base.html
|   |   |-- index.html
|   |   |-- ...
|   |
|   |-- __init__.py
|   |-- routes.py
|   |-- models.py
|   |-- forms.py
|   |-- ...
|
|-- config.py
|-- requirements.txt
|-- run.py
|-- .env (optional, for environment variables)
|-- venv/ (virtual environment directory, created when using virtualenv)
